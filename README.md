async def poll_parser():
    # –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–∏–π –∏–º–ø–æ—Ä—Ç: fname –±–µ–∑ .py
    parser_module = importlib.import_module("src.parsers." + cfg["parser_file"])

            entries = await parse_fn()

        await asyncio.sleep(interval)
